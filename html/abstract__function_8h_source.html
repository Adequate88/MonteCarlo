244template <typename InputType, typename OutputType>
245class ExponentialFunction : public AbstractFunction<InputType, OutputType> {
246public:
253 ExponentialFunction(const AbstractFunction<InputType, OutputType>& baseFunction, int power)
254 : baseFunction_(baseFunction), power_(static_cast<double>(power)), isIntegerPower_(true) {}
255
262 ExponentialFunction(const AbstractFunction<double, double>& baseFunction, double power)
263 : baseFunction_(baseFunction), power_(power), isIntegerPower_(false) {}
264
271 OutputType eval(const InputType& x) const override {
272 if (isIntegerPower_) {
273 // Use repeated multiplication for integer power
274 if (static_cast<int>(power_) == 1) {
275 return baseFunction_.eval(x);
276 }
277 OutputType baseValue = baseFunction_.eval(x);
278 OutputType result = baseFunction_.eval(x);
279 for (int i = 0; i < static_cast<int>(power_); ++i) {
280 result *= baseValue;
281 }
282 return result;
283 }
284 // Use std::pow for double power
285 const double baseValue = baseFunction_.eval(static_cast<double>(x));
286 return static_cast<OutputType>(std::pow(baseValue, power_));
287 }
288
289private:
290 const AbstractFunction<InputType, OutputType>& baseFunction_;
291 double power_;
292 bool isIntegerPower_;
293};
