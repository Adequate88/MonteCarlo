36

39 std::cout << "Mean of Original Distribution: " << 0 << std::endl;
40 std::cout << "Standard Deviation of Sample Means: " << standard_deviation_of_means << std::endl;
41 std::cout << "According to CLT, the Standard Deviation should be approximately: " << 1.0 / std::sqrt(N_) << std::endl;
42 }
