115std::vector<Token> tokenize(const std::string &expr) {
116 std::vector<Token> tokens;
117 size_t pos = 0;
118
119 auto skipWhitespace = [&]( ) {
120 while (pos < expr.size() && std::isspace(static_cast<unsigned char>(expr[pos]))) {
121 pos++;
122 }
123 };
124
125 auto readNumber = [&]( ) {
126 size_t start = pos;
127 bool seenDot = false;
128 while (pos < expr.size() && (std::isdigit(static_cast<unsigned char>(expr[pos])) || expr[pos] == '.')) {
129 if (expr[pos] == '.') {
130 if (seenDot) {
131 throw std::runtime_error("Invalid number with multiple decimal points");
132 }
133 seenDot = true;
134 }
135 pos++;
136 }
137 tokens.push_back({TokenType::NUMBER, expr.substr(start, pos - start)});
138 };
139
140 auto readIdent = [&]( ) {
141 size_t start = pos;
142 while (pos < expr.size() && std::isalpha(static_cast<unsigned char>(expr[pos]))) {
143 pos++;
144 }
145 tokens.push_back({TokenType::IDENT, expr.substr(start, pos - start)});
146 };
147
148 while (pos < expr.size()) {
149 skipWhitespace();
150 if (pos >= expr.size()) break;
151
152 char c = expr[pos];
153
154 if (std::isalpha(static_cast<unsigned char>(c))) {
155 // Identifier (function or variable)
156 readIdent();
157 } else if (std::isdigit(static_cast<unsigned char>(c)) || c == '.') {
158 // Number
159 readNumber();
160 } else {
161 // Single-char tokens
162 switch (c) {
163 case '+':
164 tokens.push_back({TokenType::PLUS, "+"});
165 pos++;
166 break;
167 case '-':
168 tokens.push_back({TokenType::MINUS, "-"});
169 pos++;
170 break;
171 case '*':
172 tokens.push_back({TokenType::MUL, "*"});
173 pos++;
174 break;
175 case '/':
176 tokens.push_back({TokenType::DIV, "/"});
177 pos++;
178 break;
179 case '^':
180 tokens.push_back({TokenType::POW, "^"});
181 pos++;
182 break;
183 case '(':
184 tokens.push_back({TokenType::LPAREN, "("});
185 pos++;
186 break;
187 case ')':
188 tokens.push_back({TokenType::RPAREN, ")"});
189 pos++;
190 break;
191 default:
192 throw std::runtime_error(std::string("Unexpected character: ") + c);
193 }
194 }
195 }
196
197 tokens.push_back({TokenType::END, ""});
198 return tokens;
199}

290 void composeFunc(const ASTNode* token) {
291 std::unique_ptr<AbstractFunction<InputType, OutputType>> func;
292
293 if (token->value == "cos") {
294 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
295 [](InputType x) { return std::cos(x); }
296 );
297 } else if (token->value == "sin") {
298 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
299 [](InputType x) { return std::sin(x); }
300 );
301 } else if (token->value == "tan") {
302 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
303 [](InputType x) { return std::tan(x); }
304 );
305 } else if (token->value == "exp") {
306 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
307 [](InputType x) { return std::exp(x); }
308 );
309 } else if (token->value == "log") {
310 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
311 [](InputType x) {
312 if (x <= 0) {
313 throw std::domain_error("Logarithm undefined for non-positive values");
314 }
315 return std::log(x);
316 }
317 );
318 } else {
319 throw std::invalid_argument("Unknown function: " + token->value);
320 }
321
322 // Ensure there is an argument on the stack for the composed function
323 if (functionStack.empty()) {
324 throw std::runtime_error("Not enough arguments on the stack for function: " + token->value);
325 }
326
327 // Move the top argument off the stack before popping
328 auto argPtr = functionStack.top();
329 functionStack.pop();
330
331 functionMemory.push_back(std::move(func));
332 auto funcPtr = functionMemory.back().get();
333
334 // Now create the composed function using both func and arg
335 auto composed = std::make_unique<ComposedFunction<InputType, OutputType>>(*funcPtr, *argPtr);
336 functionMemory.push_back(std::move(composed));
337 functionStack.push(functionMemory.back().get());
338 }

350 void performOperation(const ASTNode* token) {
351 std::unique_ptr<AbstractFunction<InputType, OutputType>> newFunc;
352 // Extract the right operand
353 auto rightPtr = functionStack.top();
354 functionStack.pop();
355
356 auto leftPtr = functionStack.top();
357 functionStack.pop();
358
359 if (token->value == "+") {
360 newFunc = std::make_unique<AddFunction<InputType, OutputType>>(*leftPtr, *rightPtr);
361
362 } else if (token->value == "*") {
363 newFunc = std::make_unique<MultiplyFunction<InputType, OutputType>>(*leftPtr, *rightPtr);
364 } else {
365 throw std::invalid_argument("Unsupported operator: " + token->value);
366 }
367
368 functionMemory.push_back(std::move(newFunc));
369 functionStack.push(functionMemory.back().get());;
370
371 }

383 void createNumerics(const ASTNode* token) {
384 std::unique_ptr<AbstractFunction<InputType, OutputType>> func;
385
386 if(token->nodeType == NodeType::NUM) {
387 InputType value;
388 if constexpr (std::is_floating_point<InputType>::value) {
389 value = static_cast<InputType>(std::stod(token->value)); // Convert to floating-point
390 } else if constexpr (std::is_integral<InputType>::value) {
391 value = static_cast<InputType>(std::stoi(token->value)); // Convert to integer
392 } else {
393 throw std::invalid_argument("Unsupported InputType for conversion");
394 }
395 func = std::make_unique<ConstantFunction<InputType>>(value);
396
397 } else if(token->nodeType == NodeType::VAR) {
398 func = std::make_unique<IdentityFunction<InputType>>();
399
400 } else {
401 throw std::invalid_argument("Invalid Token Found: " + token->value);
402 }
403
404 functionMemory.push_back(std::move(func));
405 // Push a raw pointer onto the stack
406 functionStack.push(functionMemory.back().get());
407 }

482 // Power := FunctionCall | Variable | Number | '(' Expression ')'
483 std::unique_ptr<ASTNode> parsePower() {
484 if (current().type == TokenType::IDENT) {
485 std::string name = current().value;
486 // Check if it's a function call
487 if (peek().type == TokenType::LPAREN) {
488 advance(); // consume IDENT
489 advance(); // consume '('
490 auto arg = parseExpression();
491 expect(TokenType::RPAREN);
492 advance(); // consume ')'
493 return makeFunc(name, std::move(arg));
494 } else {
495 // Just a variable
496 advance();
497 return std::make_unique<ASTNode>(NodeType::VAR, name);
498 }
499 } else if (current().type == TokenType::NUMBER) {
500 std::string val = current().value;
501 advance();
502 return std::make_unique<ASTNode>(NodeType::NUM, val);
503 } else if (current().type == TokenType::LPAREN) {
504 advance(); // consume '('
505 auto node = parseExpression();
506 expect(TokenType::RPAREN);
507 advance(); // consume ')'
508 return node;
509 } else {
510 throw std::runtime_error("Unexpected token while parsing power");
511 }
512 }
