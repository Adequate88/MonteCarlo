104 void composeFunc(const ASTNode* token) {
105 std::unique_ptr<AbstractFunction<InputType, OutputType>> func;
106
107 if (token->value == "cos") {
108 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
109 [](InputType x) { return std::cos(x); }
110 );
111 } else if (token->value == "sin") {
112 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
113 [](InputType x) { return std::sin(x); }
114 );
115 } else if (token->value == "tan") {
116 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
117 [](InputType x) { return std::tan(x); }
118 );
119 } else if (token->value == "exp") {
120 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
121 [](InputType x) { return std::exp(x); }
122 );
123 } else if (token->value == "log") {
124 func = std::make_unique<FunctionWrapper<InputType, OutputType>>(
125 [](InputType x) {
126 if (x <= 0) {
127 throw std::domain_error("Logarithm undefined for non-positive values");
128 }
129 return std::log(x);
130 }
131 );
132 } else {
133 throw std::invalid_argument("Unknown function: " + token->value);
134 }
135
136 // Ensure there is an argument on the stack for the composed function
137 if (functionStack.empty()) {
138 throw std::runtime_error("Not enough arguments on the stack for function: " + token->value);
139 }
140
141 // Move the top argument off the stack before popping
142 auto argPtr = functionStack.top();
143 functionStack.pop();
144
145 functionMemory.push_back(std::move(func));
146 auto funcPtr = functionMemory.back().get();
147
148 // Now create the composed function using both func and arg
149 auto composed = std::make_unique<ComposedFunction<InputType, OutputType>>(*funcPtr, *argPtr);
150 functionMemory.push_back(std::move(composed));
151 functionStack.push(functionMemory.back().get());
152 }

164 void performOperation(const ASTNode* token) {
165 std::unique_ptr<AbstractFunction<InputType, OutputType>> newFunc;
166 // Extract the right operand
167 auto rightPtr = functionStack.top();
168 functionStack.pop();
169
170 auto leftPtr = functionStack.top();
171 functionStack.pop();
172
173 if (token->value == "+") {
174 newFunc = std::make_unique<AddFunction<InputType, OutputType>>(*leftPtr, *rightPtr);
175 } else if (token->value == "-") {
176 newFunc = std::make_unique<SubtractFunction<InputType, OutputType>>(*leftPtr, *rightPtr);
177 } else if (token->value == "*") {
178 newFunc = std::make_unique<MultiplyFunction<InputType, OutputType>>(*leftPtr, *rightPtr);
179 } else if (token->value == "/") {
180 newFunc = std::make_unique<DivideFunction<InputType, OutputType>>(*leftPtr, *rightPtr);
181 }else if (token->value == "^") {
182 newFunc = std::make_unique<ExponentialFunction<InputType, OutputType>>(*leftPtr, *rightPtr);
183 } else {
184 throw std::invalid_argument("Unsupported operator: " + token->value);
185 }
186
187 functionMemory.push_back(std::move(newFunc));
188 functionStack.push(functionMemory.back().get());;

202 void createNumerics(const ASTNode* token) {
203 std::unique_ptr<AbstractFunction<InputType, OutputType>> func;
204
205 if(token->nodeType == NodeType::NUM) {
206 InputType value;
207 if constexpr (std::is_floating_point<InputType>::value) {
208 value = static_cast<InputType>(std::stod(token->value)); // Convert to floating-point
209 } else if constexpr (std::is_integral<InputType>::value) {
210 value = static_cast<InputType>(std::stoi(token->value)); // Convert to integer
211 } else {
212 throw std::invalid_argument("Unsupported InputType for conversion");
213 }
214 func = std::make_unique<ConstantFunction<InputType>>(value);
215
216 } else if(token->nodeType == NodeType::VAR) {
217 if (token->value[0] == '-') {
218 func = std::make_unique<NegativeIdentityFunction<InputType>>();
219 } else {
220 func = std::make_unique<IdentityFunction<InputType>>();
221 }
222 } else {
223 throw std::invalid_argument("Invalid Token Found: " + token->value);
224 }
225
226 functionMemory.push_back(std::move(func));
227 // Push a raw pointer onto the stack
228 functionStack.push(functionMemory.back().get());
229 }

304 // Power := FunctionCall | Variable | Number | '(' Expression ')'
305 std::unique_ptr<ASTNode> parsePower() {
306 if (current().type == TokenType::IDENT) {
307 std::string name = current().value;
308 // Check if it's a function call
309 if (peek().type == TokenType::LPAREN) {
310 advance(); // consume IDENT
311 advance(); // consume '('
312 auto arg = parseExpression();
313 expect(TokenType::RPAREN);
314 advance(); // consume ')'
315 return makeFunc(name, std::move(arg));
316 } else {
317 // Just a variable
318 advance();
319 return std::make_unique<ASTNode>(NodeType::VAR, name);
320 }
321 } else if (current().type == TokenType::NUMBER) {
322 std::string val = current().value;
323 advance();
324 return std::make_unique<ASTNode>(NodeType::NUM, val);
325 } else if (current().type == TokenType::LPAREN) {
326 advance(); // consume '('
327 auto node = parseExpression();
328 expect(TokenType::RPAREN);
329 advance(); // consume ')'
330 return node;
331 } else if (current().value == "-") {
332 std::string val = current().value;
333 advance();
334 val += current().value;
335 if (current().type == TokenType::NUMBER) {
336 advance();
337 return std::make_unique<ASTNode>(NodeType::NUM, val);
338 } else if (current().type == TokenType::IDENT) {
339 advance();
340 return std::make_unique<ASTNode>(NodeType::VAR, val);
341 } else {
342 throw std::runtime_error("Unexpected token while parsing negative");
343 }
344
345 } else {

347 throw std::runtime_error("Unexpected token while parsing power");
