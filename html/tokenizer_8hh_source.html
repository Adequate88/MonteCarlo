111inline std::vector<Token> tokenize(const std::string &expr) {
112 std::vector<Token> tokens;
113 size_t pos = 0;
114
115 auto skipWhitespace = [&]( ) {
116 while (pos < expr.size() && std::isspace(static_cast<unsigned char>(expr[pos]))) {
117 pos++;
118 }
119 };
120
121 auto readNumber = [&]( ) {
122 size_t start = pos;
123 bool seenDot = false;
124 while (pos < expr.size() && (std::isdigit(static_cast<unsigned char>(expr[pos])) || expr[pos] == '.')) {
125 if (expr[pos] == '.') {
126 if (seenDot) {
127 throw std::runtime_error("Invalid number with multiple decimal points");
128 }
129 seenDot = true;
130 }
131 pos++;
132 }
133 tokens.push_back({TokenType::NUMBER, expr.substr(start, pos - start)});
134 };
135
136 auto readIdent = [&]( ) {
137 size_t start = pos;
138 while (pos < expr.size() && std::isalpha(static_cast<unsigned char>(expr[pos]))) {
139 pos++;
140 }
141 tokens.push_back({TokenType::IDENT, expr.substr(start, pos - start)});
142 };
143
144 while (pos < expr.size()) {
145 skipWhitespace();
146 if (pos >= expr.size()) break;
147
148 char c = expr[pos];
149
150 if (std::isalpha(static_cast<unsigned char>(c))) {
151 // Identifier (function or variable)
152 readIdent();
153 } else if (std::isdigit(static_cast<unsigned char>(c)) || c == '.') {
154 // Number
155 readNumber();
156 } else {
157 // Single-char tokens
158 switch (c) {
159 case '+':
160 tokens.push_back({TokenType::PLUS, "+"});
161 pos++;
162 break;
163 case '-':
164 tokens.push_back({TokenType::MINUS, "-"});
165 pos++;
166 break;
167 case '*':
168 tokens.push_back({TokenType::MUL, "*"});
169 pos++;
170 break;
171 case '/':
172 tokens.push_back({TokenType::DIV, "/"});
173 pos++;
174 break;
175 case '^':
176 tokens.push_back({TokenType::POW, "^"});
177 pos++;
178 break;
179 case '(':
180 tokens.push_back({TokenType::LPAREN, "("});
181 pos++;
182 break;
183 case ')':
184 tokens.push_back({TokenType::RPAREN, ")"});
185 pos++;
186 break;
187 default:
188 throw std::runtime_error(std::string("Unexpected character: ") + c);
189 }
190 }
191 }
192
193 tokens.push_back({TokenType::END, ""});
194 return tokens;
195}
